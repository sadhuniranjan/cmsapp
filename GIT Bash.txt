cd :- change directory 
	cd e:/
	cd folderName/folderName1
	cd .. ---> (/e/folderName)
	cd folderName1/ -->(/e/folderName/folderName1)
	cd ../.. -->(/e)

Ctrl+a :- Move Cursor to the beginning.
Ctrl+e :- Move Cursor to the end.
Ctrl+u :- Delete to the beginning from cursor.
Ctrl+k :- Delete to the end from Cursor.

ls         :- Display files.(Short Format)
ls -l      :- Displau files.(Long Format)
ls -l r*   :- Display filer starting with r.
ls -a      :- Display all files.(including hidden files)
ls -la/-al :- Display all files.(in long format)
